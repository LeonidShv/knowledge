4-callback function

callback function - send it's into inother function, and in body other function, we execute callback

      1) callback for array method

      let arr = [1,2,3,4]

      function doM(item) {
          return item * 2;
      }

      console.log( arr.map(doM) )

      2) just example

      function greeting(name) {
          alert('Hello ' + name);
        }
        
        function processUserInput(callback) {
          var name = prompt('Please enter your name.');
          callback(name);
        }
        
        processUserInput(greeting);
