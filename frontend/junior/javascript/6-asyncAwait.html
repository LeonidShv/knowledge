async await - an async function that returns promise

await - you can use only inside an async function, it waits for finishing the fetch request.

async await - it's a new way for writing async code, before the same functionality made using promise 

    async function f() {

        let promise = new Promise((resolve, reject) => {
          setTimeout(() => resolve("готово!"), 1000)
        });
      
        let result = await promise; // будет ждать, пока промис не выполнится (*)
      
        alert(result); // "готово!"
      }
