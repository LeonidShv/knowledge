closures

closures - oportunity to return body of funciton

        let arr = [1,2,3,4]

        function doM(size) {
            return function(item) {
                return item * size;
            };
        };

        var size2 = doM(2);
        var size3 = doM(3);

        arr.map(size2);   [2, 4, 6, 8]
        arr.map(size3);  [3, 6, 9, 12]
